# ml/disagreement_resolver.py
class DisagreementResolver:
    def __init__(self):
        self.confidence_weights = {
            'agentR': 0.6,      # Rule-based: reliable but rigid
            'ml_model': 0.8,     # ML: learns from data
            'historical_accuracy': 0.9  # Past performance
        }
    
    def resolve(self, agentR_score, ml_score, context_history):
        disagreement_threshold = 0.2  # 20% difference
        
        if abs(agentR_score - ml_score) > disagreement_threshold:
            # Conflict detected - apply resolution rules
            
            # Get historical accuracy for both
            agentR_acc = self.get_historical_accuracy('agentR')
            ml_acc = self.get_historical_accuracy('ml')
            
            # Winner determined by: confidence * accuracy * recency
            if (self.confidence_weights['ml_model'] * ml_acc) > \
               (self.confidence_weights['agentR'] * agentR_acc):
                winner = 'ML'
                reason = "ML has higher historical accuracy"
            else:
                winner = 'AgentR'
                reason = "Rule-based more consistent with validation data"
            
            return {
                'final_score': (agentR_score + ml_score) / 2 if winner == 'ML' else agentR_score,
                'disagreement_resolved_as': winner,
                'resolution_reason': reason,
                'confidence_interval': self.calculate_ci(agentR_score, ml_score)
            }
Resolution Strategies
Minor Disagreement (<30%): Average the scores
Moderate Disagreement (30-50%): Weight by historical accuracy
Major Disagreement (>50%): Escalate to human review + flag for validation study

# ml/explainability.py
class ExplainabilityLayer:
    def explain_risk(self, user_data, ml_prediction, agent_outputs):
        explanation = {
            'primary_factors': [],
            'supporting_evidence': [],
            'confidence_score': ml_prediction['confidence'],
            'evidence_sources': []
        }
        
        # Top contributing features
        for feature, weight in ml_prediction['top_features'][:3]:
            explanation['primary_factors'].append({
                'factor': feature,
                'weight': round(weight, 3),
                'clinical_significance': self.get_clinical_meaning(feature)
            })
        
        # Map to DSM-5 criteria
        explanation['evidence_sources'] = [
            f"DSM-5 {criterion}: {present}" 
            for criterion, present in ml_prediction['dsm5_matches'].items()
        ]
        
        # Agent agreement/disagreement
        explanation['agent_consensus'] = self.calculate_consensus(
            agent_outputs, ml_prediction
        )
        
        return explanation
